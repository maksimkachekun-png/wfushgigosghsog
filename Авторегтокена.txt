import asyncio
import re
import sqlite3
import os
from aiogram import Bot, Dispatcher, types
from aiogram.filters import Command
from pymax import SocketMaxClient
from pymax.models import UserAgentPayload

# --- НАСТРОЙКИ ---
API_TOKEN = "8787042752:AAGGFTxbo0q1j1i9CD7PPTOE5FDKNDXOHbY"
WORK_DIR = "sessions"

bot = Bot(token=API_TOKEN)
dp = Dispatcher()

sessions: dict[int, "SessionData"] = {}

class SessionData:
    def __init__(self, phone: str, client: SocketMaxClient):
        self.phone = phone
        self.client = client
        self.temp_token = None
        self.awaiting_password = False
        self.last_code = None

def get_work_dir(phone: str) -> str:
    path = os.path.join(WORK_DIR, phone.strip("+"))
    os.makedirs(path, exist_ok=True)
    return path

async def create_client(phone: str) -> SocketMaxClient:
    ua = UserAgentPayload(device_type="DESKTOP", app_version="25.12.13")
    client = SocketMaxClient(phone=phone, work_dir=get_work_dir(phone), headers=ua)
    await client.connect()
    return client

def read_token_from_db(phone: str) -> str | None:
    db_path = os.path.join(get_work_dir(phone), "session.db")
    if not os.path.exists(db_path):
        return None
    try:
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT token FROM sessions LIMIT 1")
        row = cursor.fetchone()
        conn.close()
        return row[0] if row else None
    except:
        return None

# --- КОМАНДЫ ---

@dp.message(Command("start"))
async def cmd_start(message: types.Message):
    await message.answer(
        "🤖 Бот для получения токена Max\n\n"
        "/login — начать авторизацию\n"
        "/cancel — отменить сессию"
    )

@dp.message(Command("cancel"))
async def cmd_cancel(message: types.Message):
    user_id = message.from_user.id
    if user_id in sessions:
        try:
            await sessions[user_id].client.disconnect()
        except:
            pass
        sessions.pop(user_id, None)
        await message.answer("❌ Сессия отменена.")
    else:
        await message.answer("Нет активной сессии.")

@dp.message(Command("login"))
async def cmd_login(message: types.Message):
    user_id = message.from_user.id
    if user_id in sessions and sessions[user_id] is not None:
        await message.answer("Уже есть активная сессия. Сначала /cancel")
        return
    sessions[user_id] = None
    await message.answer("📱 Отправь номер в формате +79XXXXXXXXX")

# --- ОБРАБОТКА СООБЩЕНИЙ ---

@dp.message()
async def handle_message(message: types.Message):
    user_id = message.from_user.id
    text = message.text.strip()

    # --- Облачный пароль ---
    if user_id in sessions:
        s = sessions[user_id]
        if isinstance(s, SessionData) and s.awaiting_password:
            await message.answer("🔐 Вхожу с облачным паролем...")
            try:
                # Пробуем оба варианта метода
                if hasattr(s.client, 'login_with_cloud_password'):
                    await s.client.login_with_cloud_password(s.temp_token, text)
                elif hasattr(s.client, 'login_with_code_password'):
                    await s.client.login_with_code_password(s.temp_token, s.last_code, text)
                else:
                    await s.client.login_with_code(s.temp_token, s.last_code, text)
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
                return
            await finish_login(message, s)
            return

    # --- Номер телефона ---
    if re.fullmatch(r"\+7\d{10}", text):
        if user_id in sessions and sessions[user_id] is None:
            phone = text
            await message.answer(f"⏳ Запрашиваю код для {phone}...")
            try:
                client = await create_client(phone)
                temp_token = await client.request_code(phone)
                session = SessionData(phone=phone, client=client)
                session.temp_token = temp_token
                sessions[user_id] = session
                await message.answer(
                    f"✅ Код отправлен на {phone}\n\n"
                    "✏️ Пришли мне код (6 цифр)"
                )
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
                sessions.pop(user_id, None)
        else:
            await message.answer("Сначала /login")
        return

    # --- Код из СМС ---
    code_match = re.fullmatch(r"(\d{6})", text)
    if code_match and user_id in sessions:
        s = sessions[user_id]
        if not isinstance(s, SessionData):
            await message.answer("Сначала /login")
            return
        code = code_match.group(1)
        s.last_code = code
        await message.answer("🔑 Вхожу...")
        try:
            resp = await s.client.login_with_code(s.temp_token, code)
            if isinstance(resp, dict) and "password" in str(resp).lower():
                s.awaiting_password = True
                await message.answer("🔐 Нужен облачный пароль. Отправь:")
                return
            await finish_login(message, s)
        except Exception as e:
            if "password" in str(e).lower():
                s.awaiting_password = True
                await message.answer("🔐 Нужен облачный пароль. Отправь:")
            else:
                await message.answer(f"❌ Ошибка: {e}")
        return

    await message.answer("Ожидаю номер или код.")

async def finish_login(message: types.Message, s: SessionData):
    token = read_token_from_db(s.phone) or getattr(s.client, 'token', None)
    if token:
        await message.answer(f"✅ Успешно!\n\n<code>{token}</code>", parse_mode="HTML")
    else:
        await message.answer(f"⚠️ Токен не найден. Проверь папку sessions/{s.phone.strip('+')}")
    try:
        await s.client.disconnect()
    except:
        pass
    sessions.pop(message.from_user.id, None)

async def main():
    os.makedirs(WORK_DIR, exist_ok=True)
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
