WORK_DIR = "sessions"

bot = Bot(token=API_TOKEN)
dp = Dispatcher()

# Хранилище сессий: user_id -> список активных сессий
# Каждая сессия — это dict с состоянием, НЕ привязана к user_id глобально
sessions: dict[str, "SessionData"] = {}  # ключ — уникальный session_id

class SessionData:
    def __init__(self, phone: str, client: SocketMaxClient, token_type: str, user_id: int):
        self.phone = phone
        self.client = client
        self.token_type = token_type
        self.user_id = user_id
        self.temp_token = None
        self.awaiting_password = False
        self.last_code = None

def get_work_dir(phone: str) -> str:
    path = os.path.join(WORK_DIR, phone.strip("+"))
    os.makedirs(path, exist_ok=True)
    return path

async def create_client(phone: str, token_type: str) -> SocketMaxClient:
    real_type = "DESKTOP" if token_type == "DESKTOP_CONSOLE" else "WEB"
    ua = UserAgentPayload(device_type=real_type, app_version="25.12.13")
    client = SocketMaxClient(phone=phone, work_dir=get_work_dir(phone), headers=ua)
    await client.connect()
    return client

def read_token_from_db(phone: str) -> str | None:
    db_path = os.path.join(get_work_dir(phone), "session.db")
    if not os.path.exists(db_path):
        return None
    try:
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT token FROM sessions LIMIT 1")
        row = cursor.fetchone()
        conn.close()
        return row[0] if row else None
    except:
        return None

def read_viewer_id_from_db(phone: str) -> int | None:
    db_path = os.path.join(get_work_dir(phone), "session.db")
    if not os.path.exists(db_path):
        return None
    try:
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        for col in ["user_id", "viewer_id", "viewerId"]:
            try:
                cursor.execute(f"SELECT {col} FROM sessions LIMIT 1")
                row = cursor.fetchone()
                if row and row[0]:
                    conn.close()
                    return row[0]
            except:
                continue
        conn.close()
        return None
    except:
        return None

def build_console_code(token: str, viewer_id: int | None = None) -> str:
    if not viewer_id:
        viewer_id = 134819431
    device_id = str(uuid.uuid4())

    code = (
        'sessionStorage.clear();\n'
        'localStorage.clear();\n'
        f"localStorage.setItem('__oneme_device_id', '{device_id}');\n"
        f"localStorage.setItem('__oneme_auth', JSON.stringify({{\"token\":\"{token}\",\"viewerId\":{viewer_id}}}));\n"
        'window.location.reload();'
    )
    return code

# Храним, какой user_id какой session_id ждёт (для команд)
user_states: dict[int, str] = {}  # user_id -> session_id или "choosing_type"

type_keyboard = ReplyKeyboardMarkup(
    keyboard=[
        [KeyboardButton(text="🖥 DESKTOP (консоль)")],

@dp.message(Command("start"))
async def cmd_start(message: types.Message):
    await message.answer(

@dp.message(Command("cancel"))
async def cmd_cancel(message: types.Message):
    user_id = message.from_user.id
    cancelled = 0
    to_remove = []
    for sid, s in sessions.items():
        if s.user_id == user_id:
            try:
                await s.client.disconnect()
            except:
                pass
            to_remove.append(sid)
            cancelled += 1
    for sid in to_remove:
        sessions.pop(sid, None)
    user_states.pop(user_id, None)
    
    if cancelled > 0:
        await message.answer(f"❌ Отменено сессий: {cancelled}")
    else:
        await message.answer("Нет активных сессий.")

@dp.message(Command("list"))
async def cmd_list(message: types.Message):
    user_id = message.from_user.id
    active = []
    for sid, s in sessions.items():
        if s.user_id == user_id:
            status = "ожидает пароль" if s.awaiting_password else "в процессе"
            active.append(f"• {s.phone} ({s.token_type}) — {status}")
    
    if active:
        await message.answer("Активные сессии:\n" + "\n".join(active))
    else:
        await message.answer("Нет активных сессий.")

@dp.message(Command("login"))
async def cmd_login(message: types.Message):
    user_id = message.from_user.id
    user_states[user_id] = "choosing_type"
    await message.answer(
        "📱 Выбери формат:",

    )

@dp.message()
async def handle_message(message: types.Message):
    user_id = message.from_user.id
    text = message.text.strip()

    # --- ЭТАП 0: Выбор формата ---
    if text in ["🖥 DESKTOP (консоль)", "💻 WEB (консоль)"]:
        if user_states.get(user_id) == "choosing_type":
            token_type = "DESKTOP_CONSOLE" if text == "🖥 DESKTOP (консоль)" else "WEB_CONSOLE"
            
            # Создаём новый session_id
            session_id = f"{user_id}_{uuid.uuid4().hex[:8]}"
            user_states[user_id] = session_id
            sessions[session_id] = {"token_type": token_type, "awaiting_phone": True, "user_id": user_id}
            
            await message.answer(
                f"✅ Выбран: {text}\n\n"
                "📱 Теперь отправь номер в формате +79XXXXXXXXX",
                reply_markup=types.ReplyKeyboardRemove()
            )
            return

    if user_states.get(user_id) == "choosing_type":
        await message.answer("Сначала выбери формат кнопкой")
        return

    # --- Ищем активную сессию для этого пользователя по номеру ---
    # Если пришёл номер телефона — это НОВАЯ сессия
    # Если пришёл код или пароль — ищем сессию, которая ждёт ввода

    # --- ЭТАП 1: Облачный пароль ---
    for sid, s in list(sessions.items()):
        if isinstance(s, SessionData) and s.user_id == user_id and s.awaiting_password:
            await message.answer("🔐 Вхожу с облачным паролем...")
            try:
                if hasattr(s.client, 'login_with_cloud_password'):
                    await s.client.login_with_cloud_password(s.temp_token, text)
                elif hasattr(s.client, 'login_with_code_password'):
                    await s.client.login_with_code_password(s.temp_token, s.last_code, text)
                else:
                    await s.client.login_with_code(s.temp_token, s.last_code, text)
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
                return
            await finish_login(message, s, sid)
            return

    # --- ЭТАП 2: Номер телефона (новая сессия) ---
    if re.fullmatch(r"\+7\d{10}", text):
        # Ищем session_id, который ждёт номер
        state = user_states.get(user_id)
        if state and state in sessions and isinstance(sessions[state], dict) and sessions[state].get("awaiting_phone"):
            session_id = state
            token_type = sessions[session_id]["token_type"]
            phone = text
            
            await message.answer(f"⏳ Запрашиваю код для {phone}...")
            try:
                client = await create_client(phone, token_type)
                temp_token = await client.request_code(phone)
                session = SessionData(phone=phone, client=client, token_type=token_type, user_id=user_id)
                session.temp_token = temp_token
                sessions[session_id] = session
                user_states[user_id] = session_id  # обновляем состояние
                await message.answer(
                    f"✅ Код отправлен на {phone}\n\n"
                    "✏️ Пришли мне код (6 цифр)\n\n"
                    "ℹ️ Можешь сразу запускать следующий номер через /login"
                )
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
                sessions.pop(session_id, None)
                user_states.pop(user_id, None)
        else:
            await message.answer("Сначала /login и выбери формат")
        return

    # --- ЭТАП 3: Код из СМС ---
    code_match = re.fullmatch(r"(\d{6})", text)
    if code_match:
        code = code_match.group(1)
        # Ищем сессию, которая ждёт код (не ждёт пароль, не ждёт номер)
        found = False
        for sid, s in list(sessions.items()):
            if isinstance(s, SessionData) and s.user_id == user_id and not s.awaiting_password and s.last_code is None:
                s.last_code = code
                await message.answer("🔑 Вхожу...")
                try:
                    resp = await s.client.login_with_code(s.temp_token, code)
                    if isinstance(resp, dict) and "password" in str(resp).lower():
                        s.awaiting_password = True
                        await message.answer("🔐 Нужен облачный пароль. Отправь:")
                        return
                    await finish_login(message, s, sid)
                except Exception as e:
                    if "password" in str(e).lower():
                        s.awaiting_password = True
                        await message.answer("🔐 Нужен облачный пароль. Отправь:")
                    else:
                        await message.answer(f"❌ Ошибка: {e}")
                found = True
                break
        
        if not found:
            await message.answer("Нет сессии, ожидающей код. Сначала отправь номер или /login")
        return

    await message.answer("Ожидаю номер, код или пароль.")

async def finish_login(message: types.Message, s: SessionData, session_id: str):
    token = read_token_from_db(s.phone) or getattr(s.client, 'token', None)

    if not token:
        await message.answer(f"⚠️ Токен не найден. Проверь папку sessions/{s.phone.strip('+')}")
    else:
        viewer_id = read_viewer_id_from_db(s.phone)
        code = build_console_code(token, viewer_id)

        label = "десктоп-приложения" if s.token_type == "DESKTOP_CONSOLE" else "веб-версии"
        await message.answer(
            f"✅ <b>Готово!</b> ({s.phone})\n\n"
            f"Код для консоли {label} Max:\n\n"
            f"<code>{code}</code>",
            parse_mode="HTML"
        )

    try:
        await s.client.disconnect()
    except:
        pass
    sessions.pop(session_id, None)
    # Убираем состояние пользователя, если нет других активных сессий
    has_other = any(s for sid, s in sessions.items() if isinstance(s, SessionData) and s.user_id == message.from_user.id)
    if not has_other:
        user_states.pop(message.from_user.id, None)

async def main():
    os.makedirs(WORK_DIR, exist_ok=True)
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
