import pymax
# Находим все публичные имена в pymax
_names = [x for x in dir(pymax) if not x.startswith('_')]

def _find_class(part_name):
    """Ищет класс, содержащий part_name в названии"""
    for name in _names:
        if part_name.lower() in name.lower():
            return getattr(pymax, name)
    raise ImportError(f"Не могу найти класс содержащий '{part_name}' в pymax. Доступные: {_names}")

try:
    # Пробуем угадать имена классов
    ClientClass = _find_class('socket') or _find_class('client')
except:
    # Если угадать не получилось, пробуем частые варианты
    for cls_name in ['SocketMaxClient', 'Client', 'MaxClient']:
        if hasattr(pymax, cls_name):
            ClientClass = getattr(pymax, cls_name)
            break
    else:
        raise ImportError(f"Не нашел класс клиента в pymax. Доступные объекты: {_names}")

try:
    UAClass = _find_class('user') or _find_class('agent')
except:
    for cls_name in ['UserAgentPayload', 'UserAgent']:
        if hasattr(pymax, cls_name):
            UAClass = getattr(pymax, cls_name)
            break
    else:
        UAClass = None  # Необязательный класс

# Хранилище активных сессий
sessions: dict[int, "SessionData"] = {}

class SessionData:
    def __init__(self, phone: str, client):
        self.phone = phone
        self.client = client
        self.temp_token = None
        self.awaiting_password = False
        self.last_code = None

def get_work_dir(phone: str) -> str:
    """Создаёт и возвращает путь к папке сессии для номера"""
    path = os.path.join(WORK_DIR, phone.strip("+"))
    os.makedirs(path, exist_ok=True)
    return path

async def create_client(phone: str):
    """Создаёт и подключает клиент PyMax"""
    if UAClass:
        ua = UAClass(device_type="DESKTOP", app_version="25.12.13")
        client = ClientClass(phone=phone, work_dir=get_work_dir(phone), headers=ua)
    else:
        client = ClientClass(phone=phone, work_dir=get_work_dir(phone))
    await client.connect()
    return client

def read_token_from_db(phone: str) -> str | None:
    """Читает токен из session.db"""
    db_path = os.path.join(get_work_dir(phone), "session.db")
    if not os.path.exists(db_path):
        return None
    try:
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT token FROM sessions LIMIT 1")
        row = cursor.fetchone()
        conn.close()
        return row[0] if row else None
    except Exception:
        return None

@dp.message(Command("cancel"))
async def cmd_cancel(message: types.Message):
    user_id = message.from_user.id
    if user_id in sessions:
        try:
            await sessions[user_id].client.disconnect()
        except:
            pass
        sessions.pop(user_id, None)
        await message.answer("❌ Сессия отменена.")
    else:
        await message.answer("Нет активной сессии.")

@dp.message(Command("login"))
async def cmd_login(message: types.Message):
    user_id = message.from_user.id
    if user_id in sessions and sessions[user_id] is not None:
        await message.answer("У тебя уже есть активная сессия. Сначала /cancel")
        return
    sessions[user_id] = None
    await message.answer("📱 Отправь номер телефона в формате +79XXXXXXXXX")

@dp.message()
async def handle_message(message: types.Message):
    user_id = message.from_user.id
    text = message.text.strip()

    # --- ЭТАП 1: Ожидание облачного пароля ---
    if user_id in sessions:
        session = sessions[user_id]
        if isinstance(session, SessionData) and session.awaiting_password:
            password = text
            await message.answer("🔐 Пытаюсь войти с облачным паролем...")
            try:
                # Пробуем разные названия методов
                for method_name in ['login_with_cloud_password', 'login_with_code_password']:
                    if hasattr(session.client, method_name):
                        await getattr(session.client, method_name)(
                            temp_token=session.temp_token,
                            password=password
                        )
                        break
            except Exception as e:
                await message.answer(f"❌ Ошибка: {e}")
                return
            await finish_login(message, session)
            return

    # --- ЭТАП 2: Ввод номера телефона ---
    if re.fullmatch(r"\+7\d{10}", text):
        if user_id in sessions and sessions[user_id] is None:
            phone = text
            await message.answer(f"⏳ Подключаюсь и запрашиваю код для {phone}...")
            try:
                client = await create_client(phone)
                # Ищем метод запроса кода
                for method_name in ['request_code', 'send_code']:
                    if hasattr(client, method_name):
                        temp_token = await getattr(client, method_name)(phone)
                        break
                else:
                    raise AttributeError("Нет метода request_code или send_code")
                
                session = SessionData(phone=phone, client=client)
                session.temp_token = temp_token
                sessions[user_id] = session
                
                await message.answer(
                    f"✅ Код отправлен на {phone}\n\n"
                    "✏️ Пришли мне код из СМС (6 цифр)\n\n"
                    "⚠️ Если на аккаунте есть облачный пароль, "
                    "бот сам запросит его после ввода кода."
                )
            except Exception as e:
                await message.answer(f"❌ Ошибка при запросе кода: {e}")
                sessions.pop(user_id, None)
        else:
            await message.answer("Сначала введи /login")
        return

    # --- ЭТАП 3: Ввод кода из СМС ---
    code_match = re.fullmatch(r"(\d{6})", text)
    if code_match and user_id in sessions:
        session = sessions[user_id]
        if not isinstance(session, SessionData):
            await message.answer("Сначала введи /login")
            return

        code = code_match.group(1)
        session.last_code = code
        
        await message.answer("🔑 Выполняю вход...")
        
        try:
            # Ищем метод для входа с кодом
            resp = None
            for method_name in ['login_with_code', 'confirm_code']:
                if hasattr(session.client, method_name):
                    resp = await getattr(session.client, method_name)(
                        session.temp_token, code
                    )
                    break
            
            # Проверяем ответ на требование пароля
            if isinstance(resp, dict):
                if resp.get("status") == "cloud_password_required" or \
                   resp.get("error") == "cloud_password_required":
                    session.awaiting_password = True
                    await message.answer(
                        "🔐 Этот аккаунт защищён облачным паролем!\n"
                        "Отправь мне облачный пароль:"
                    )
                    return
            
            await finish_login(message, session)
            
        except Exception as e:
            error_msg = str(e).lower()
            if "password" in error_msg or "cloud" in error_msg:
                session.awaiting_password = True
                await message.answer(
                    "🔐 Этот аккаунт защищён облачным паролем!\n"
                    "Отправь мне облачный пароль:"
                )
            else:
                await message.answer(f"❌ Ошибка авторизации: {e}")
        return

    await message.answer(
        "🤔 Не понял. Ожидаю:\n"
        "• Номер (+79XXXXXXXXX) — после /login\n"
        "• Код из СМС (6 цифр)\n"
        "• Или облачный пароль (если был запрошен)"
    )

# --- ЗАВЕРШЕНИЕ ВХОДА ---

async def finish_login(message: types.Message, session: SessionData):
    user_id = message.from_user.id
    phone = session.phone
    
    token = read_token_from_db(phone)
    if not token:
        try:
            token = session.client.token
        except:
            token = None
    
    if token:
        await message.answer(
            "✅ <b>Авторизация успешна!</b>\n\n"
            f"<code>{token}</code>\n\n"
            "⚠️ Храни токен в безопасности. Не передавай третьим лицам.",
            parse_mode="HTML"
        )
    else:
        await message.answer(
            "⚠️ Вход выполнен, но токен получить не удалось.\n"
            f"Проверь файл вручную: sessions/{phone.strip('+')}/session.db"
        )
    
    try:
        await session.client.disconnect()
    except:
        pass
    
    sessions.pop(user_id, None)
